# Flutter Gallery 工程分析

入口 main函数 lib/main.dart

	// 导入库
	import 'package:flutter/material.dart';
	import 'gallery/app.dart';
	
	void main() {
	  // runApp 是 flutter 提供的全局函数
	  // https://docs.flutter.io/flutter/widgets/runApp.html
	  runApp(const GalleryApp());
	}

runApp 函数定义

	// flutter/lib/src/widgets/binding.dart
	void runApp(Widget app) {
	  WidgetsFlutterBinding.ensureInitialized()  // 确保app 实例初始化
	    ..attachRootWidget(app)  // 调用实例方法 
	    ..scheduleWarmUpFrame(); // 调用实例方法
	}
